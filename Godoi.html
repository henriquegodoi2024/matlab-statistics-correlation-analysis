% Problem 2 -> Function for mean and variance when an array is
% given for input

function v = myVariance(x)
    % Ensure x is a column vector
    x = x(:);

    % Number of elements
    n = length(x);

    % Mean
    xbar = sum(x) / n;

    % Variance
    v = sum((x - xbar).^2) / (n - 1);

end

function cov_val = myCovariance(X, Y)

    n = length(X);
    mX = sum(X)/n;
    mY = sum(Y)/n;
    cov_val = sum((X - mX) .* (Y - mY)) / (n - 1);
end

function r = myCorrelation(X, Y)
    n = length(X);
    cov_val = sum((X - mean(X)) .* (Y - mean(Y))) / (n - 1);

    std_X = sqrt(sum((X - mean(X)).^2) / (n - 1));
    std_Y = sqrt(sum((Y - mean(Y)).^2) / (n - 1));

    r = cov_val / (std_X * std_Y);
end
